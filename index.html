import pandas as pd

sales = pd.read_csv("sales.csv")
summary = (
    sales
    .assign(revenue=lambda d: d.units * d.price)
    .query("region != 'Test'")
    .groupby("category", as_index=False)
    .agg(revenue=("revenue", "sum"))
    .sort_values("revenue", ascending=False)
)

print(summary.head())
